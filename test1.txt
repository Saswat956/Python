import numpy as np

def five_number_summary(data):
    """
    Calculate the five-number summary of a dataset.

    Args:
    data (list or numpy array): The dataset for which to calculate the summary.

    Returns:
    dict: A dictionary containing the minimum, first quartile (Q1), median (Q2), third quartile (Q3), and maximum.
    """
    # Ensure the data is sorted
    sorted_data = np.sort(data)

    # Calculate quartiles
    q1 = np.percentile(sorted_data, 25)
    q2 = np.median(sorted_data)
    q3 = np.percentile(sorted_data, 75)

    # Calculate minimum and maximum
    min_val = sorted_data[0]
    max_val = sorted_data[-1]

    # Create a dictionary for the summary
    summary = {
        'Minimum': min_val,
        'Q1 (First Quartile)': q1,
        'Median (Q2)': q2,
        'Q3 (Third Quartile)': q3,
        'Maximum': max_val
    }

    return summary

# Example usage:
data = [12, 16, 18, 22, 24, 25, 28, 30, 32, 36, 42]
result = five_number_summary(data)
print(result)
