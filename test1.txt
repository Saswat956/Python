import matplotlib.pyplot as plt
import seaborn as sns

# Assuming you have two datasets as lists or NumPy arrays
data1 = [15, 20, 25, 30, 35, 40, 45, 50, 55, 60, 65]
data2 = [25, 28, 30, 32, 35, 38, 40, 42, 45, 48, 50]

# Create a figure and axis
fig, ax = plt.subplots()

# Plot the distributions using seaborn
sns.histplot(data1, kde=True, color='blue', label='Dataset 1', ax=ax)
sns.histplot(data2, kde=True, color='red', label='Dataset 2', ax=ax)

# Add labels and a title
plt.xlabel("Values")
plt.ylabel("Density")
plt.title("Overlay Distribution Plot of Two Datasets")

# Add a legend
plt.legend()

# Show the plot
plt.show()
