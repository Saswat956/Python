import matplotlib.pyplot as plt
import numpy as np

# Assuming you have two datasets as lists or NumPy arrays
data1 = [15, 20, 25, 30, 35, 40, 45, 50, 55, 60, 65]
data2 = [25, 28, 30, 32, 35, 38, 40, 42, 45, 48, 50]

# Create a figure and axis
fig, ax = plt.subplots()

# Create a box plot for the two datasets side by side
boxplot = ax.boxplot([data1, data2], labels=['Dataset 1', 'Dataset 2'])

# Add labels and a title
ax.set_xlabel("Datasets")
ax.set_ylabel("Values")
ax.set_title("Box Plot Comparison of Two Datasets")

# Calculate summary statistics
stats1 = np.percentile(data1, [0, 25, 50, 75, 100])
stats2 = np.percentile(data2, [0, 25, 50, 75, 100])

# Annotate the plot with summary statistics
for i, stat in enumerate(stats1):
    ax.text(i + 1, stat, f'{stat:.2f}', va='bottom', ha='center', fontsize=10, color='blue')

for i, stat in enumerate(stats2):
    ax.text(i + 2, stat, f'{stat:.2f}', va='bottom', ha='center', fontsize=10, color='red')

# Show the plot
plt.show()
